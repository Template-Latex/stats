if (!String.format) {
    String.format = function ($format) {
        let $args = Array.prototype.slice.call(arguments, 1);
        return $format.replace(/{(\d+)}/g, function ($match, $number) {
            return typeof $args[$number] !== 'undefined' ?
                $args[$number] :
                $match;
        });
    };
}
